            document.addEventListener('DOMContentLoaded', () => {
        const educationRadio = document.getElementById('option1');
        const experienceRadio = document.getElementById('option2');
        const educationContainer = document.getElementById('education-container');
        const experienceContainer = document.getElementById('experience-container');

        // Fungsi untuk menampilkan kontainer yang sesuai
        const showContainer = (containerToShow, containerToHide) => {
            // Sembunyikan kontainer yang tidak dipilih
            containerToHide.classList.add('hidden');
            
            // Tampilkan kontainer yang dipilih
            containerToShow.classList.remove('hidden');
        };

        educationRadio.addEventListener('change', () => {
            if (educationRadio.checked) {
                showContainer(educationContainer, experienceContainer);
            }
        });

        experienceRadio.addEventListener('change', () => {
            if (experienceRadio.checked) {
                showContainer(experienceContainer, educationContainer);
            }
        });
    });
